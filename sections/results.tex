\subsection{B6 (\SI{3x2}{} Ganging)}
\begin{frame}{II6-B6 - Efficiencies @ PSI}

	\subfigs{\subfig{.45}{EM1}[Efficiency Map].}{\subfig{.45}[r]{EffVol}[Efficiency vs. voltage.]}
	
	\begin{itemize}\itemfill
